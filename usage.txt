# FIFA World Cup 2026 — simulator scripts
# Default data files: data/ratings.csv and data/scores.csv


# ── main.py ──────────────────────────────────────────────────────────────────
# Monte Carlo tournament simulator

uv run main.py                                    # 10,000 simulations, text output
uv run main.py -n 500                             # quick test
uv run main.py -n 10000 --output json             # JSON output for piping
uv run main.py --trace Switzerland                # one-run match diary for a team

#   -n N / --simulations N   number of simulations (default: 10000)
#   --output text|json        output format (default: text)
#   --trace TEAM              one-run diary for TEAM; ignores -n and --output
#   --ratings FILE            Glicko-2 ratings CSV (default: data/ratings.csv)
#   --scores FILE             completed scores CSV (default: data/scores.csv)


# ── visualize.py ─────────────────────────────────────────────────────────────
# Heatmap of round-reach probabilities; reads JSON from main.py

uv run main.py -n 1000 --output json | uv run visualize.py
uv run main.py -n 1000 --output json | uv run visualize.py --save chart.png
uv run main.py -n 1000 --output json | uv run visualize.py --top 20
uv run visualize.py results.json                  # from a saved JSON file

#   [FILE]       JSON file from main.py --output json (default: stdin)
#   --top N      show only the top N teams by win probability (default: 48)
#   --save FILE  save chart to file instead of opening a window


# ── matchup.py ───────────────────────────────────────────────────────────────

uv run matchup.py Germany France
uv run matchup.py Germany France -n 50000

uv run matchup.py "Cote d'Ivoire" Ecuador -n 5000
uv run matchup.py USA Mexico --scores data/scores.csv --ratings data/ratings.csv
uv run matchup.py --today                         # today's schedule with Glicko-2 odds

#   team_a team_b           two team names (required unless --today)
#   -n N / --simulations N  matches to simulate (default: 10,000)
#   --today                 show today's schedule with win/draw/loss odds; no simulation
#   --save FILE             save chart to file instead of opening a window
#   --ratings FILE          Glicko-2 ratings CSV (default: data/ratings.csv)
#   --scores FILE           completed scores CSV (default: data/scores.csv)

# ── update_scores.py ─────────────────────────────────────────────────────────
# Fetch completed results from ESPN and update scores.csv (including match date)

uv run update_scores.py                           # fetch today's results
uv run update_scores.py --days 3                  # scan last 3 days
uv run update_scores.py --dry-run                 # preview without writing
uv run update_scores.py --days 30 --dry-run       # preview full backfill
# Note: if any existing scores are missing a date, the script automatically
#       scans from tournament start to backfill all missing dates.

#   --days N       days to scan back from today (default: 1 = today only)
#   --scores FILE  scores CSV to update (default: data/scores.csv)
#   --dry-run      preview changes without writing to disk

# ── timeline.py ──────────────────────────────────────────────────────────────
# Line chart of each team's simulated ranking (or win probability) across match days
# Requires update_scores.py to have been run (needs date column in scores.csv)
# Results are cached per match day; stale entries are detected and re-simulated
# automatically (no manual cache deletion needed after code updates).

uv run timeline.py
uv run timeline.py -n 5000
uv run timeline.py --win-prob                     # y-axis = win probability (%) instead of rank
uv run timeline.py --teams "France,Argentina,Germany"
uv run timeline.py --save timeline.png
uv run timeline.py --no-cache                     # force re-simulation of all entries

# --snapshot: two-panel slope chart (no timeline dates required)
#   Left panel  — Glicko-2 rating: pre-tournament (left) → current (right)
#   Right panel — tournament win %: pre-tournament (left) → current (right)
#   Rising line = better than expected (green gradient, darker = bigger gain)
#   Falling line = worse than expected (red gradient, darker = bigger drop)
#   Flat line = as expected (gray); team names on the outer edge of each panel
#   Caching: baseline stored under "_baseline"; current stored under the latest
#   match date; --no-cache recomputes both but preserves other cache entries.

uv run timeline.py --snapshot
uv run timeline.py --snapshot --save snapshot.png
uv run timeline.py --snapshot --no-cache          # force fresh simulation

#   --snapshot              two-panel before/after slope chart; skips timeline
#   --win-prob              y-axis = win probability (%) instead of rank (timeline only)
#   -n N / --simulations N  simulations per match day / snapshot (default: 2000)
#   --teams TEAMS           comma-separated list of teams to highlight (timeline only)
#   --save FILE             save chart to PNG instead of displaying
#   --cache FILE            cache file path (default: cache/timeline_N.json)
#   --no-cache              force re-simulation; preserves unaffected cache entries
#   --ratings FILE          Glicko-2 ratings CSV (default: data/ratings.csv)
#   --scores FILE           completed scores CSV (default: data/scores.csv)
